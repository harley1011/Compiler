program
{
    int x = 5;
    int inputs[100];

    for (int i = 0; i < x; i = i + 1) {
        get(inputs[i]);
    };

    int add;
    int k;
    add = 5;

    k = addValueToArray(inputs, x, add);

    for (int t = 0; t < x; t = t + 1) {
            put(inputs[t]);
    };
};

int addValueToArray(int inputs[100], int x, int add) {
    for (int i = 0; i < x; i = i + 1 ) {
        inputs[i] = inputs[i] + add;
    };

    return (0);
};
